# Usage Guide

Learn how to use Day Flow in various scenarios.

## Basic Calendar

The simplest way to use the calendar:

```tsx
import { useCalendarApp, DayFlowCalendar, createMonthView } from 'dayflow';

export default function MyCalendar() {
  const calendar = useCalendarApp({
    views: [createMonthView()],
    initialDate: new Date(),
  });

  return <DayFlowCalendar calendar={calendar} />;
}
```

## With State Management

Control the calendar with your own state:

```tsx
import { useState } from 'react';
import { useCalendarApp, DayFlowCalendar, createMonthView } from 'dayflow';

export default function MyCalendar() {
  const [date, setDate] = useState(new Date());

  const calendar = useCalendarApp({
    views: [createMonthView()],
    initialDate: date,
    callbacks: {
      onDateChange: newDate => {
        setDate(newDate);
      },
    },
  });

  return <DayFlowCalendar calendar={calendar} />;
}
```

## Event Handling

Handle user interactions:

```tsx
import {
  useCalendarApp,
  DayFlowCalendar,
  createMonthView,
  Event,
} from 'dayflow';

export default function MyCalendar() {
  const handleDateClick = (date: Date) => {
    console.log('Date clicked:', date);
  };

  const calendar = useCalendarApp({
    views: [createMonthView()],
    initialDate: new Date(),
    callbacks: {
      onDateChange: handleDateClick,
      onEventCreate: (event: Event) => {
        console.log('Event created:', event);
      },
    },
  });

  return <DayFlowCalendar calendar={calendar} />;
}
```

## Custom Events

Display custom events on the calendar:

```tsx
import {
  useCalendarApp,
  DayFlowCalendar,
  createMonthView,
  Event,
} from 'dayflow';

const events: Event[] = [
  {
    id: '1',
    title: 'Meeting',
    start: new Date(2024, 9, 10, 10, 0),
    end: new Date(2024, 9, 10, 11, 0),
    color: '#3b82f6',
  },
  {
    id: '2',
    title: 'Conference',
    start: new Date(2024, 9, 15),
    end: new Date(2024, 9, 17),
    allDay: true,
    color: '#10b981',
  },
];

export default function MyCalendar() {
  const calendar = useCalendarApp({
    views: [createMonthView()],
    initialDate: new Date(),
    events,
  });

  return <DayFlowCalendar calendar={calendar} />;
}
```

## Different Views

Switch between month, week, and day views:

```tsx
import {
  useCalendarApp,
  DayFlowCalendar,
  createDayView,
  createWeekView,
  createMonthView,
  ViewType,
} from 'dayflow';

export default function MyCalendar() {
  const calendar = useCalendarApp({
    views: [createDayView(), createWeekView(), createMonthView()],
    defaultView: ViewType.MONTH,
    initialDate: new Date(),
  });

  return <DayFlowCalendar calendar={calendar} />;
}
```

## Customization

Customize the calendar appearance:

```tsx
import { useCalendarApp, DayFlowCalendar, createMonthView } from 'dayflow';

export default function MyCalendar() {
  const calendar = useCalendarApp({
    views: [createMonthView()],
    initialDate: new Date(),
  });

  return (
    <div
      style={
        {
          '--primary-color': '#3b82f6',
          '--bg-color': '#ffffff',
          '--text-color': '#1f2937',
        } as React.CSSProperties
      }
    >
      <DayFlowCalendar calendar={calendar} />
    </div>
  );
}
```
